let start= datetime(@fromDate); 
let end= datetime(@toDate);
let SREIncidents =()
{   
        Incidents
        | where CreateDate >= start and CreateDate <= end and OccurringEnvironment =="PROD" and OwningTeamName in("SKYPEFORBUSINESS\\IC3ProvisioningTriage","SKYPEFORBUSINESS\\RTCAPPSRE")
        | summarize arg_max(ModifiedDate ,*) by IncidentId        
        | project IncidentId 
 };
  SREIncidents 
        | join  
        ( Incidents  ) on IncidentId
| project IncidentId ,OwningTeamName,ModifiedDate
| order by IncidentId,ModifiedDate asc
